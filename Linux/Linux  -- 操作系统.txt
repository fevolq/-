查看目录
ls 路径位置 
ls -l   #查看文件属性  - 文件   d 目录  l 链接文件
ls -a   #查看所有文件 包括隐藏文件
ls -la  #查看所有的文件的文件属性

路径位置
	绝对路径：从根目录开始到文件本身的完整路径
	相对路径：文件本身  相对当前位置

切换目录
cd 路径位置
cd ..  #切换到上一层目录
cd   #切换到当前用户的家目录
cd -  #相邻目录的切换(针对操作)
cd ../../../  #连升三级


查看当前位置的绝对路径
pwd

目录
  创建目录
  mkdir 目录名
  mkdir -p 目录/目录/目录  #创建连接目录

  删除目录
  rmdir 目录名  #只能删除空目录
  rm -rf 目录名   r递归删除目录(先内再外)  f  强制执行无提示 

  复制
  cp -r 目录 目标目录
  目标目录不存在则会新建

  剪切/重命名
  mv 目录 目标目录
  目标目录不存在则重命名


文件
	touch 文件名  #创建文件
	rm 文件名    #删除文件
	cp 文件  目标    #复制文件
	mv 文件 目标    #剪切\重命名

查看文件内容
	cat 文件名  #从第一行开始显示文件所有内容
	more  文件名  #分页查看  空格键翻页  Ctrl+b\f 上下翻页  自动退出
	less  文件名  #分页查看  空格键翻页  Ctrl+b\f 上下翻页  按q退出
	head -n 值 文件  #查看文件前几行内容 默认10行
	tail -n 值 文件  #查看文件后几行内容 
	tail -f 文件名  #动态查看

重定向、追加
>  重定向  会覆盖原有内容
>> 追加    不会覆盖原有内容

命令 > 文件名

echo 内容 >  
		 >>

查找
	find 路径 -name 文件名
	find 路径 -name '*文件名*'  

链接文件
	软链接
	ln -s 源文件 链接文件
	相当于快捷方式，文件内容会同步
	删除源文件会影响链接文件，删除链接文件不会影响源文件


	硬链接
	ln 源文件 链接文件
